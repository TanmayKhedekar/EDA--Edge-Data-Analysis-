
# AZURE TEXT ANALYTICS SETUP GUIDE (BY YASH TAGUNDE)

## STEP 1 — CREATE AZURE LANGUAGE RESOURCE
az cognitiveservices account create \
  --name TanmayTextAnalytics \
  --resource-group IoTEdgeResources \
  --kind TextAnalytics \
  --sku F0 \
  --location southeastasia

# Get keys
az cognitiveservices account keys list \
  --name TanmayTextAnalytics \
  --resource-group IoTEdgeResources -o json

## STEP 2 — SET ENVIRONMENT VARIABLES
export AZURE_LANGUAGE_KEY="YOUR_KEY_HERE" #Put your api key here
export AZURE_LANGUAGE_ENDPOINT="YOUR_ENDPOINT_HERE"

## STEP 3 — INSTALL DEPENDENCIES
pip install azure-ai-textanalytics

## STEP 4 — CREATE SAMPLE TEXT FILE
echo "Azure Cognitive Services are amazing and make AI easy to use." > Sample_text.txt

## STEP 5 — RUN SENTIMENT ANALYSIS
python3 TextSentimentAnalysis.py sample_text.txt --out Output_Sentiment.txt

## STEP 6 — VIEW OUTPUT
cat Output_Sentiment.txt
